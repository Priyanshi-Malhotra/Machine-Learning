Sex : Male or Female

Age : Age of the passenger

Fare : Passenger Fare

Cabin : Cabin

def fill_age(cols):
    Class=cols[0]
    Age=cols[1]
    
    if pd.isnull(Age):
        if Class==1:
            return 39
        elif Class==2:
            return 32
        else:
            return 28
    return Age
